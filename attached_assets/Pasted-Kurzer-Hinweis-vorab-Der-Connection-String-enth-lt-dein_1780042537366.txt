  export function getBootstrapConfig() {
    try {
      if (fs.existsSync(BOOTSTRAP_PATH)) {
        const raw = fs.readFileSync(
          BOOTSTRAP_PATH, 'utf8')
        return JSON.parse(raw)
      }
    } catch { /* ignore */ }
    return { database_url: '' }
  }
  
  export function saveBootstrapConfig(
    config: { database_url: string }
  ) {
    fs.writeFileSync(
      BOOTSTRAP_PATH,
      JSON.stringify(config, null, 2),
      'utf8'
    )
  }
  
  export function getDatabaseUrl(): string {
    // Priority 1: environment variable
    if (process.env.DATABASE_URL?.trim()) {
      return process.env.DATABASE_URL.trim()
    }
    // Priority 2: bootstrap.json
    const cfg = getBootstrapConfig()
    if (cfg.database_url?.trim()) {
      return cfg.database_url.trim()
    }
    // Priority 3: hardcoded default
    return DEFAULT_DB_URL
  }

STEP 5 — Replace all SQLite query syntax:

  Find every db.prepare(...).run(...)
  Find every db.prepare(...).get(...)  
  Find every db.prepare(...).all()
  
  Replace with async pool.query() calls:

  Pattern translations:
    ? placeholders → $1, $2, $3
    .get(params) → (await query(sql, params)).rows[0]
    .all(params) → (await query(sql, params)).rows
    .run(params) → await query(sql, params)
    lastInsertRowid → use RETURNING id:
      INSERT INTO ... RETURNING id
      → result.rows[0].id
