BACKEND: new settings routes

  GET /api/admin/settings/:group
    Admin only. group = 'ai' | 'mail' | 'delivery'
    Returns settings for that group 
    (NEVER return API keys in plain text — 
    return masked value: "••••••••" + 
    last 4 chars if key is set, "" if empty)

  PATCH /api/admin/settings/:group
    Admin only.
    Body: { key: value, ... }
    For API keys: only update if the sent 
    value is NOT the masked placeholder 
    (i.e. does not contain "••••••••").
    This prevents accidentally overwriting 
    a key with the display mask.
    Returns { success: true }

  POST /api/admin/settings/test-mail
    Admin only.
    Sends a test email using current mail 
    settings to the admin's own email address.
    Subject: "GAIO Analyzer — Mailserver-Test"
    Body: "Der Mailserver ist korrekt 
    konfiguriert."
    Returns { success: true } or 
    { success: false, error: string }

  GET /api/admin/settings/ai-status
    Admin only.
    Returns which providers have API keys 
    configured (boolean per provider) and 
    which provider is currently active.
    Also returns which provider the app is 
    currently using for its analyses.

BACKEND: email sending module

Create server/mailer.js (or .ts):

  import nodemailer from 'nodemailer'
  
  export async function getTransporter(db) {
    const settings = getMailSettings(db)
    if (!settings.host) return null
    return nodemailer.createTransport({
      host: settings.host,
      port: parseInt(settings.port),
      secure: settings.secure === 'true',
      auth: {
        user: settings.user,
        pass: settings.password
      }
    })
  }
  
  export async function sendMail(db, options) {
    const transporter = await getTransporter(db)
    const settings = getMailSettings(db)
    
    if (!transporter) {
      // Log to console as fallback
      console.log('[MAIL FALLBACK]', {
        to: options.to,
        subject: options.subject,
        text: options.text
      })
      return { success: true, fallback: true }
    }
    
    await transporter.sendMail({
      from: `"${settings.fromName}" 
        <${settings.fromAddress}>`,
      to: options.to,
      subject: options.subject,
      html: options.html || options.text,
      bcc: options.bcc || undefined
    })
    return { success: true, fallback: false }
  }

Update all existing email sends (welcome 
email, verification code) to use this 
mailer instead of console.log.
