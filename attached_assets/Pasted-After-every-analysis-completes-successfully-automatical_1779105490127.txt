After every analysis completes successfully, 
automatically generate the HTML export and 
store it in the analysis log — silently, 
without any user interaction or visible 
feedback.

CHANGE: trigger silent background export

At that point, immediately call the existing 
HTML export generation function — the same 
function used when the user clicks 
"Report als HTML exportieren".

The difference: do NOT trigger a file 
download. Instead, capture the generated 
HTML string and POST it to the log endpoint.

Implement as follows:

  useEffect(() => {
    if (!report || report.status !== 'completed')
      return
    
    // Fire and forget — do not await, 
    // do not show loading, do not show errors 
    // to the user
    autoSaveHtmlExport(report)
  }, [report?.status])

  async function autoSaveHtmlExport(report) {
    try {
      // Call the same HTML generation logic 
      // already used for the download button,
      // but return the string instead of 
      // triggering a download.
      // If the generator is currently a 
      // function that directly triggers 
      // download, refactor it to accept an 
      // optional parameter:
      //   generateHtmlReport(report, 
      //     { download: false }) 
      //     → returns html string
      //   generateHtmlReport(report, 
      //     { download: true }) 
      //     → triggers file download (existing behavior)
      
      const htmlString = generateHtmlReport(
        report, { download: false }
      )
      
      if (!htmlString) return
      
      // POST to backend — use the analysis 
      // log id if available, otherwise let 
      // the backend create a new log entry
      await fetch('/api/admin/analysis-log/auto-export', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          htmlContent: htmlString,
          domain: report.url,
          companyName: report.companyName || '',
          gaioScore: report.gaioScore,
          scoresJson: JSON.stringify(report.scores),
          pagesCrawled: report.crawledPages?.length ?? 0,
        })
      })
      // Silently succeed or fail — no user feedback
    } catch {
      // Silently ignore all errors — 
      // this must never affect the user experience
    }
  }
